Before running below commands,
make sure .env and application.properties file (use docker db service credentials) is properly configured.
then rebuild war. and proceed with below commands.

docker rmi vote_ease_app
docker rmi mchiir/vote_ease_app
docekr rmi postgres:18


docker compose down -v

docker compose up -d --build
docker logs -f vote_ease_snapshot

docker compose up -d app
docker compose up -d db

docker compose stop <running_container_ID>

docker tag vote_ease_app mchiir/vote_ease_app:latest
docker push mchiir/vote_ease_app:latest
